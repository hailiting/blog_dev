# js 异步 for 循环

```js
let dishes = [
  { name: "fish", time: 1 },
  { name: "fish1", time: 2 },
  { name: "fish3", time: 3 },
];
(async () => {
  for (let d of dishes) {
    console.log("开始做" + d.name);
    await (() => {
      return new Promise((res) => {
        setTimeout(res, d.time * 1000);
      });
    })();
    console.log("做好了");
  }
})();
```
