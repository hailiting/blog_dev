## 合约验证

- 编码格式：EIP191 / EIP712
  - EIP191: 区分交易签名和其他信息签名
    - 0x19 初始字节：这个初始字节确保signed_data不是有效的RLP编码，从而防止签名数据呗误解为以太坊交易
    - <1 byte version> 可以自行定义签名数据版本，占用一字节，可以用0
    - <version sepific data>： 不定长的消息头数据
    - <data to sign>：原始签名数据;
      - `[0x19][ 0x45=hex("E")] ["thereum Signed Message:\n || len(message)] [message]"`
  - EIP712: 定义结构化信息签名标准
- Keccak256（**凯恰克**） 哈希算法，是SHA-3族：安全性高，灵活，速度快
- 椭圆曲线加密算法 Secp256kl: 高效性，安全性，标准化

contract EIP712Verifier is EIP712 {
  using ECDSA for bytes32;
  struct send {
    address to;
    uint256 value;
  }
  bytes public constant SEND_TYPEHASH = keccak256("Send(address to,Uint256 value)");
  constructor() EIP712("EIP712Verifier", "1.0.0"){}
  function hashSend(Send memory send) public view returns (bytes32) {
    return _hashTypedDataV4(keccak256(abi.encode(SEND_TYPEHASH, send.to, send.value)));
  }
  function verify(address signer, Send memory send, bytes memory signature) public view returns (bool) {
    bytes32 digest = hashSend(send);
    return digest.recover(signature) == signer;
  }
}

### ERC20-Permit eip2612

- 签名授权
  - (授权)可以在线下签名进行，签名信息可以在执行接收转账交易时提交到链上（permit），让授权和转账在一笔交易里完成
  - 同事转账交易也可以由接收方（或其他第三方）来提交，也避免了用户（ERC20的拥有者）需要有ETH的依赖
- 签名要注意的
  - 签名被重用
  - 如何防止重用，签名中加入
    - Nonce
    - ChainID
    - Deadline
    - 业务逻辑+授权(amount:, spender)

```solidity
function permit(address owner, address spender, uint256 value, uint256 deadline, uint8 v, bytes32 r, bytes32 s) public virtual {
  if(block.timestamp > deadline) {
    revert ERC2612ExpiredSignature(deadline);
  }
  bytes32 structHash = keccak256(abi.encode(PERMIT_TYPEHASH, owner, spender, value, _useNonce(owner), deadline));
  bytes32 hash = _hashTypedDataV4(structHash);
  address signer = ECDSA.recover(hash, v, r, s);
  if(signer != owner){
    revert ERC2612InvalidSignature(signer, owner);
  }
  _approve(owner, spender, value);
}
```

const USDC = '0xA0b86991c6218b36c1d19D4a2e9Eb0cE3606eB48';
const AAVE_POOL = '0x87870Bca3F3fD6335C3F4ce8392D69350B4fA4E2';
let signer, usdcContract, aaveContract;

// 初始化
async function init() {
  if (window.ethereum) {
    await window.ethereum.request({ method: 'eth_requestAccounts' });
    const provider = new ethers.providers.Web3Provider(window.ethereum);
    signer = provider.getSigner();
    const userAddress = await signer.getAddress();

    // 显示余额
    const usdcAbi = ['function balanceOf(address) view returns (uint256)'];
    usdcContract = new ethers.Contract(USDC, usdcAbi, provider);
    const balance = await usdcContract.balanceOf(userAddress);
    document.getElementById('balance').innerText = ethers.utils.formatUnits(balance, 6);
  }
}

// 🎯 核心：一键存款函数
async function depositWithPermit() {
  const amount = '0.01';
  const amountInWei = ethers.utils.parseUnits(amount, 6); // USDC 是6位小数
  const userAddress = await signer.getAddress();

  // 1. 获取 nonce
  const usdcPermitAbi = ['function nonces(address) view returns (uint256)'];
  const usdcPermitContract = new ethers.Contract(USDC, usdcPermitAbi, signer);
  const nonce = await usdcPermitContract.nonces(userAddress);

  // 2. 设置过期时间（1小时后）
  const deadline = Math.floor(Date.now() / 1000) + 3600;

  // 3. 构建签名数据
  const domain = {
    name: 'USD Coin', // 代币名称
    version: '2', // 代币版本
    chainId: 1, // 以太坊主网
    verifyingContract: USDC,
  };

  const types = {
    Permit: [
      { name: 'owner', type: 'address' },
      { name: 'spender', type: 'address' },
      { name: 'value', type: 'uint256' },
      { name: 'nonce', type: 'uint256' },
      { name: 'deadline', type: 'uint256' },
    ],
  };

  const value = {
    owner: userAddress,
    spender: AAVE_POOL, // 授权给AAVE池
    value: amountInWei.toString(),
    nonce: nonce.toString(),
    deadline: deadline,
  };

  // 4. 弹出钱包签名请求！用户在这里确认
  console.log('请查看钱包弹窗...');
  const signature = await signer._signTypedData(domain, types, value);

  // 5. 拆分签名
  const sig = ethers.utils.splitSignature(signature);

  // 6. 调用AAVE的 depositWithPermit
  const aaveAbi = [
    'function supplyWithPermit(address asset, uint256 amount, address onBehalfOf, uint16 referralCode, uint256 deadline, uint8 v, bytes32 r, bytes32 s) external',
  ];
  aaveContract = new ethers.Contract(AAVE_POOL, aaveAbi, signer);

  const tx = await aaveContract.supplyWithPermit(
    USDC, // 资产地址
    amountInWei, // 金额
    userAddress, // 存款受益人
    0, // 推荐码
    deadline, // 过期时间
    sig.v,
    sig.r,
    sig.s // 签名
  );

  console.log('交易已发送:', tx.hash);
  alert('存款成功！交易哈希: ' + tx.hash);
}

init();
