-> 固定的输入一定是固定的输出
-> 缓存
-> 柯里化

```js
/// 柯里化之前
function add(x, y) {
  return x + y;
}
add(1, 2);
/// 柯里化之后
function addX(x) {
  return function(y) {
    return x + y;
  };
}
var add8 = addX(8); // 被缓存了
var x = add8(9);
```

```js
function foo(p1, p2) {
  this.val = p1 + p2;
}
var bar = foo.bind(null, "p1");
var baz = new bar("p2");
console.log(baz.val);
```
